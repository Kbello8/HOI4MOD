                ARG = { has_war_with = COL }

                ARG = { has_war_with = PAR }
                ARG = { has_war_with = ECU }
            }
        }
	}

            factor = 20

                factor = 10

            }
            modifier = {
                factor = .5
                
                NOT = { BRA = { is_in_faction_with = USA } }
            }
			modifier = {
				factor = 10

					tag = ARG

				}
			}

		name = arg.8.a
