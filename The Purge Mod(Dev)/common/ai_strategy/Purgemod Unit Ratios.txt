infantry_limit = {
	enable = {
		AND = {
			has_army_manpower = { size > 500000 }
			has_equipment = { infantry_equipment < 10000 }
		}
	}
	abort = {
		NOT = {
			AND = {
				has_army_manpower = { size > 500000 }
				has_equipment = { infantry_equipment < 10000 }
			}
		}
	}
	ai_strategy = { type = build_army id = infantry                              value = -100 }
	ai_strategy = { type = build_army id = garrison                              value = -100 }

}
