italy_egypt = {
	enable = {
		original_tag = ITA
        has_government = fascism
        EGY = { is_subject_of  = ENG }
		ITA = { has_war_with = ENG }
		ITA = { has_war_with = EGY }
	}
	abort = {
		OR = {
			NOT = { country_exists = EGY }
			OR = {
				ITA = { controls_state = 446 }
				any_country = {
					is_in_faction_with = ITA
					controls_state = 446
				}
			}
		}
	}
	
	ai_strategy = {
		type = invade
		id = "EGY"			
		value = 200
	}	
}
