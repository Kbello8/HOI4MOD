# lets make sure the allies dont overcommit to invasions early on
careful_with_early_invasions = {
	enable = {
		NOT = { has_tech = landing_craft }
		
		OR = {
			is_in_faction_with = ENG # anyone in allies
			tag = FRA
		}

		date < 1942.1.1
		NOT = {
			GER = { has_war_with = SOV }
		}
	}
	abort = {
		has_tech = landing_craft
		OR = {
			GER = { has_war_with = SOV }
			date > 1942.1.1
			is_in_faction_with = USA
		}
	}

	ai_strategy = {
		type = invade
		id = "ITA"			
		value = -500
	}

	ai_strategy = {
		type = invade
		id = "GER"			
		value = -500
	}
}

east_prussia_invasion_is_death_trap = {
	enable = {
		
		OR = {
			is_in_faction_with = ENG # anyone in allies
			tag = FRA
		}

		GER = { has_war_with = POL }
		POL = { 
			OR = {
				exists = yes 
				has_capitulated = no
			}
		}

		date < 1940.1.1
	}
	
	abort = {
		OR = {
			NOT = { GER = { has_war_with = POL } }
			POL = { 
				OR = {
					exists = no
					has_capitulated = yes
				}
			}
			date > 1940.1.1
		}
	}

	ai_strategy = {
		type = invade
		id = "GER"			
		value = -2000
	}
}
